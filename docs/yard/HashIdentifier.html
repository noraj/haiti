def samples(input)
  samples = []
  PROTOTYPES.each do |prototype|
    prototype['modes'].each do |mode|
      samples << Chf.new(mode).samples if [mode['name'], mode['hashcat'].to_s, mode['john']].include?(input)
    end
  end
  samples.delete(nil)
  samples.flatten.uniq
end
